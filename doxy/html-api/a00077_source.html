40/**
41 * Return a human-readable error string for the given libopentracedecode error
42 * code.
43 *
44 * @param error_code A libopentracedecode error code number, such as
45 * OTD_ERR_MALLOC.
46 *
47 * @return A const string containing a short, human-readable (English)
48 * description of the error, such as "memory allocation error".
49 * The string must NOT be free'd by the caller!
50 *
51 * @see otd_strerror_name
52 *
53 * @since 0.2.0
54 */
55OTD_API const char *otd_strerror(int error_code)
56{
57 const char *str;
58
59 /*
60 * Note: All defined OTD_* error macros from libopentracedecode.h must
61 * have an entry in this function, as well as in otd_strerror_name().
62 */
63
64 switch (error_code) {
65 case OTD_OK:
66 str = "no error";
67 break;
68 case OTD_ERR:
69 str = "generic/unspecified error";
70 break;
71 case OTD_ERR_MALLOC:
72 str = "memory allocation error";
73 break;
74 case OTD_ERR_ARG:
75 str = "invalid argument";
76 break;
77 case OTD_ERR_BUG:
78 str = "internal error";
79 break;
80 case OTD_ERR_PYTHON:
81 str = "Python API error";
82 break;
83 case OTD_ERR_DECODERS_DIR:
84 str = "decoders directory access error";
85 break;
86 default:
87 str = "unknown error";
88 break;
89 }
90
91 return str;
92}

94/**

99 *
100 * This function can be used for various purposes where the "name" string of
101 * a libopentracedecode error code is useful.

103 * @param error_code A libopentracedecode error code number, such as
104 * OTD_ERR_MALLOC.

106 * @return A const string containing the "name" of the error code as string.
107 * The string must NOT be free'd by the caller!

109 * @see otd_strerror

111 * @since 0.2.0
112 */

123 case OTD_OK:
124 str = "OTD_OK";
