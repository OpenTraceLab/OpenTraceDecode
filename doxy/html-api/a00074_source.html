42PyMODINIT_FUNC PyInit_opentracedecode(void)
43{
44 PyObject *mod, *Decoder_type;
45 PyGILState_STATE gstate;
46
47 gstate = PyGILState_Ensure();
48
49 mod = PyModule_Create(&opentracedecode_module);
50 if (!mod)
51 goto err_out;
52
53 Decoder_type = otd_Decoder_type_new();
54 if (!Decoder_type)
55 goto err_out;
56 if (PyModule_AddObject(mod, "Decoder", Decoder_type) < 0)
57 goto err_out;
58
59 /* Expose output types as symbols in the opentracedecode module */
60 if (PyModule_AddIntConstant(mod, "OUTPUT_ANN", OTD_OUTPUT_ANN) < 0)
61 goto err_out;
62 if (PyModule_AddIntConstant(mod, "OUTPUT_PYTHON", OTD_OUTPUT_PYTHON) < 0)
63 goto err_out;
64 if (PyModule_AddIntConstant(mod, "OUTPUT_BINARY", OTD_OUTPUT_BINARY) < 0)
65 goto err_out;
66 if (PyModule_AddIntConstant(mod, "OUTPUT_LOGIC", OTD_OUTPUT_LOGIC) < 0)
67 goto err_out;
68 if (PyModule_AddIntConstant(mod, "OUTPUT_META", OTD_OUTPUT_META) < 0)
69 goto err_out;
70 /* Expose meta input symbols. */
71 if (PyModule_AddIntConstant(mod, "OTD_CONF_SAMPLERATE", OTD_CONF_SAMPLERATE) < 0)
72 goto err_out;
73
74 mod_opentracedecode = mod;
75
76 PyGILState_Release(gstate);
77
78 return mod;
79
80err_out:
81 Py_XDECREF(mod);
82 otd_exception_catch("Failed to initialize module");
83 PyGILState_Release(gstate);
84
85 return NULL;
86}
