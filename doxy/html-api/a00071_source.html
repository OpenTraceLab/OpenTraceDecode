57/**
58 * Set the libopentracedecode loglevel.
59 *
60 * This influences the amount of log messages (debug messages, error messages,
61 * and so on) libopentracedecode will output. Using OTD_LOG_NONE disables all
62 * messages.
63 *
64 * Note that this function itself will also output log messages. After the
65 * loglevel has changed, it will output a debug message with OTD_LOG_DBG for
66 * example. Whether this message is shown depends on the (new) loglevel.
67 *
68 * @param loglevel The loglevel to set (OTD_LOG_NONE, OTD_LOG_ERR,
69 * OTD_LOG_WARN, OTD_LOG_INFO, OTD_LOG_DBG, or OTD_LOG_SPEW).
70 *
71 * @return OTD_OK upon success, OTD_ERR_ARG upon invalid loglevel.
72 *
73 * @since 0.1.0
74 */
75OTD_API int otd_log_loglevel_set(int loglevel)
76{
77 if (loglevel < OTD_LOG_NONE || loglevel > OTD_LOG_SPEW) {
78 otd_err("Invalid loglevel %d.", loglevel);
79 return OTD_ERR_ARG;
80 }
81
82 cur_loglevel = loglevel;
83
84 otd_dbg("libopentracedecode loglevel set to %d.", loglevel);
85
86 return OTD_OK;
87}

101/**
102 * Set the libopentracedecode log callback to the specified function.
103 *
104 * @param cb Function pointer to the log callback function to use.
105 * Must not be NULL.
106 * @param cb_data Pointer to private data to be passed on. This can be used
107 * by the caller to pass arbitrary data to the log functions.
108 * This pointer is only stored or passed on by libopentracedecode,
109 * and is never used or interpreted in any way. The pointer
110 * is allowed to be NULL if the caller doesn't need/want to
111 * pass any data.
112 *
113 * @return OTD_OK upon success, OTD_ERR_ARG upon invalid arguments.
114 *
115 * @since 0.3.0
116 */
117OTD_API int otd_log_callback_set(otd_log_callback cb, void *cb_data)
118{
119 if (!cb) {
120 otd_err("log: %s: cb was NULL", __func__);
121 return OTD_ERR_ARG;
122 }
123
124 /* Note: 'cb_data' is allowed to be NULL. */
125
126 otd_log_cb = cb;
127 otd_log_cb_data = cb_data;
128
129 return OTD_OK;
130}

132/**
133 * Get the libopentracedecode log callback routine and callback data.
134 *
135 * @param[out] cb Pointer to a function pointer to receive the log callback
136 * function. Optional, can be NULL.
137 * @param[out] cb_data Pointer to a void pointer to receive the log callback's
138 * additional arguments. Optional, can be NULL.
139 *
140 * @return OTD_OK upon success.
141 *
142 * @since 0.5.2
143 */
144OTD_API int otd_log_callback_get(otd_log_callback *cb, void **cb_data)
145{
146 if (cb)
147 *cb = otd_log_cb;
148 if (cb_data)
149 *cb_data = otd_log_cb_data;
150
151 return OTD_OK;
152}

154/**
155 * Set the libopentracedecode log callback to the default built-in one.
156 *
157 * Additionally, the internal 'otd_log_cb_data' pointer is set to NULL.
158 *
159 * @return OTD_OK upon success, a (negative) error code otherwise.
160 *
161 * @since 0.1.0
162 */
163OTD_API int otd_log_callback_set_default(void)
164{
165 /*
166 * Note: No log output in this function, as it should safely work
167 * even if the currently set log callback is buggy/broken.
168 */
169 otd_log_cb = otd_logv;
170 otd_log_cb_data = NULL;
171
172 return OTD_OK;
173}
174
175static int otd_logv(void *cb_data, int loglevel, const char *format,
176 va_list args)
177{
178 /* This specific log callback doesn't need the void pointer data. */
179 (void)cb_data;
180
181 (void)loglevel;
182
183 if (fputs("srd: ", stderr) < 0
184 || g_vfprintf(stderr, format, args) < 0
185 || putc('\n', stderr) < 0)
186 return OTD_ERR;
187
188 fflush(stderr);
189
190 return OTD_OK;
191}

193/** @private */
194OTD_PRIV int otd_log(int loglevel, const char *format, ...)
195{
196 int ret;
197 va_list args;
198
199 /* Only output messages of at least the selected loglevel(s). */
200 if (loglevel > cur_loglevel)
201 return OTD_OK;
202
203 va_start(args, format);
204 ret = otd_log_cb(otd_log_cb_data, loglevel, format, args);
205 va_end(args);
206
207 return ret;
208}
